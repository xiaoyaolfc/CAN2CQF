cplusplus {{
#include "inet/linklayer/ethernet/EtherFrame_m.h"
}}
// 直接从INET中扩展标准EtherNet帧
class inet::EthernetIIFrame;
namespace CAN4CQF;

//
// 定义基于IEEE802.1Qch的time-trigger message，其本质与IEEE802.1Qbv参数相同，仅在可视化方面有所区别，方便区别于原有框架代码
//
packet IEEE8021QchTTFlow extends EthernetIIFrame {
    // 无数据载荷时的payload
    byteLength = 22;
    // IEEE802.1Q的网络类型提示
    etherType = 0x8100;
    // 表示报文优先级的，TT流仅使用前两个优先级
    uint8_t pcp;
    //或者结合PCP来指示在存在拥塞的情况下有资格丢弃的帧。 
    bool de;    
    // VLAN标识符
    uint16_t VID = 0;   
    // 设置message 颜色
    //#给几个可选项：rect,oval,triangle,hexagon,circle,roundedrect
    string displayString = "b=100,100,hexagon,blue";       
}
