58 void Rxgate::handleParameterChange(const char* parname){
59 // 当GCL发生变化时，对其进行处理，判断是否为'o' OR 'C'，并根据状态发送到相应队列中
60  CoRE4INET::Scheduled::handleParameterChange(parname);
61  if (!parname || !strcmp(parname, "numGates"))//判断是否是"numGates"，如果是则更新
62  {
63  this->numGates = static_cast<unsigned int>(CoRE4INET::parameterULongCheckRange(par("numGates"), 1, std::numeric_limits<int>::max()));
64  }
65  if (!parname || !strcmp(parname, "controlList"))//判断是否是"controlList"。如果是则更新
66  {
67  gateControlList.clear();
68  vector<string> controlRows = cStringTokenizer(par("controlList"), ";").asVector();//用";"分隔controlList为门控状态元组和时间元组
69  for (vector<string>::const_iterator controlRow = controlRows.begin(); controlRow != controlRows.end(); ++controlRow)
70  {
71  vector<string> controlRowTupel = cStringTokenizer((*controlRow).c_str(), ":").asVector();//用":"分隔为门控状态
72  vector<string> controlRowGates = cStringTokenizer(controlRowTupel[0].c_str(), ",").asVector();
73  if (controlRowGates.size() != this->numGates)
74  {
75  throw cRuntimeError("controlList row size is not %d", this->numGates);
76  }
77  for (size_t i=0; i<controlRowGates.size(); i++)
78  {
79  if (strcmp(controlRowGates[i].c_str(), "o") && strcmp(controlRowGates[i].c_str(), "C"))
80  {
81  throw cRuntimeError("controlList contains unexpected character \'%s\'. Allowed are \'o\' (OPEN) or \'C\' (CLOSED).", controlRowGates[i].c_str());
82  }
83  }
84  double controlRowTime = stod(controlRowTupel[1]);//创建一个新的controlRowTime存储正确的门控时间援助
85  gateControlList.push_back(make_pair(controlRowGates, controlRowTime));// 将门控状态元组和对应的时间元组匹配创建一个新的门控列表
86  }
87  if (this->gateControlList.size() > 0)//更新相关参数
88  {
89  this->switchToFirstGateControlElement();
90  this->configNo++;
91  this->timerEventName = "Gate Control List Config " + to_string(this->configNo) + " Scheduler Event";
92  this->scheduleCurrentGateControlElementTime(false);
93  }
94  }

97 void Rxgate::handleMessage(cMessage *msg)
98 {
99  // 根据GCL判断报文发向哪个门
100  if (msg->arrivedOn("schedulerIn") && msg->getKind() == CoRE4INET::ACTION_TIME_EVENT)
101  {
102  if (!strcmp(this->timerEventName.c_str(), msg->getName()))
103  {
104  this->propagateGateControlElement(this->gateControlElement->first);
105  if (this->gateControlList.size() > 1)
106  {
107  this->switchToNextGateControlElement();
108  this->scheduleCurrentGateControlElementTime(false);
109  }
110  }
111  }
112  delete msg;
113 }

149 void Rxgate::scheduleCurrentGateControlElementTime(bool nextCycle)
150 {// 根据当前的门控元件，安排下一个门控事件。
151  CoRE4INET::SchedulerActionTimeEvent* actionTimeEvent = new CoRE4INET::SchedulerActionTimeEvent(this->timerEventName.c_str(), CoRE4INET::ACTION_TIME_EVENT);
152  uint32_t actionTime = static_cast<uint32_t>(ceil(this->gateControlElement->second / this->getOscillator()->getPreciseTick()));
153  if (actionTime >= this->getPeriod()->getCycleTicks())
154  {
155  throw cRuntimeError("The send window (%d ticks) starts outside of the period (%d ticks)",
156  actionTime, this->getPeriod()->getCycleTicks());
157  }
158  actionTimeEvent->setAction_time(actionTime);
159  actionTimeEvent->setNext_cycle(nextCycle);
160  actionTimeEvent->setDestinationGate(this->gate("schedulerIn"));
161  this->getPeriod()->registerEvent(actionTimeEvent);
162 }

164 void Rxgate::switchToFirstGateControlElement()
165 {
166  if (this->gateControlList.size() > 0)
167  {
168  this->gateControlElement = this->gateControlList.begin();
169  simtime_t currentTimeInPeriod = this->getPeriod()->getTicks() * this->getOscillator()->getPreciseTick();
170  for (auto candidateGateControlElement = this->gateControlList.begin(); candidateGateControlElement != this->gateControlList.end(); ++candidateGateControlElement)
171  {
172  if (candidateGateControlElement->second >= currentTimeInPeriod.dbl())
173  {
174  this->gateControlElement = candidateGateControlElement;
175  break;
176  }
177  }
178  if (this->gateControlElement->second != currentTimeInPeriod.dbl())
179  {
180  vector<pair<vector<string>, double>>::iterator preGateControlElement;
181  if (this->gateControlElement == this->gateControlList.begin())
182  {
183  preGateControlElement = this->gateControlList.end();
184  }
185  else
186  {
187  preGateControlElement = this->gateControlElement;
188  }
189  --preGateControlElement;
190  this->propagateGateControlElement(preGateControlElement->first);
191  }
192  }
193 }
